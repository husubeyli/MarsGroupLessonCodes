82def _find_option_with_arg(argv, short_opts=None, long_opts=None): 

83 """Search argv for options specifying short and longopt alternatives. 

84 

85 Returns: 

86 str: value for option found 

87 Raises: 

88 KeyError: if option not found. 

89 """ 

90 for i, arg in enumerate(argv): 

91 if arg.startswith('-'): 

92 if long_opts and arg.startswith('--'): 

93 name, sep, val = arg.partition('=') 

94 if name in long_opts: 

95 return val if sep else argv[i + 1] 

96 if short_opts and arg in short_opts: 

97 return argv[i + 1] 

98 raise KeyError('|'.join(short_opts or [] + long_opts or []))

117def maybe_patch_concurrency(argv=None, short_opts=None,

119 """Apply eventlet/gevent monkeypatches. 

120 

121 With short and long opt alternatives that specify the command line 

122 option to set the pool, this makes sure that anything that needs 

123 to be patched is completed as early as possible. 

124 (e.g., eventlet/gevent monkey patches). 

125 """ 

126 argv = argv if argv else sys.argv 

127 short_opts = short_opts if short_opts else ['-P']

132 pool = _find_option_with_arg(argv, short_opts, long_opts) 

133 except KeyError:

135 else: 

136 try: 

137 patcher = patches[pool] 

138 except KeyError: 

139 pass
