204def eventloop(conn, limit=None, timeout=None, ignore_timeouts=False): 

205 """Best practice generator wrapper around ``Connection.drain_events``. 

206 

207 Able to drain events forever, with a limit, and optionally ignoring 

208 timeout errors (a timeout of 1 is often used in environments where 

209 the socket can get "stuck", and is a best practice for Kombu consumers). 

210 

211 ``eventloop`` is a generator. 

212 

213 Examples: 

214 >>> from kombu.common import eventloop 

215 

216 >>> def run(conn): 

217 ... it = eventloop(conn, timeout=1, ignore_timeouts=True) 

218 ... next(it) # one event consumed, or timed out. 

219 ... 

220 ... for _ in eventloop(conn, timeout=1, ignore_timeouts=True): 

221 ... pass # loop forever. 

222 

223 It also takes an optional limit parameter, and timeout errors 

224 are propagated by default:: 

225 

226 for _ in eventloop(connection, limit=1, timeout=1): 

227 pass 

228 

229 See Also: 

230 :func:`itermessages`, which is an event loop bound to one or more 

231 consumers, that yields any messages received. 

232 """ 

233 for i in limit and range(limit) or count(): 

234 try: 

235 yield conn.drain_events(timeout=timeout) 

236 except socket.timeout: 

237 if timeout and not ignore_timeouts: # pragma: no cover 

238 raise

334def insured(pool, fun, args, kwargs, errback=None, on_revive=None, **opts): 

335 """Function wrapper to handle connection errors. 

336 

337 Ensures function performing broker commands completes 

338 despite intermittent connection failures. 

339 """ 

340 errback = errback or _ensure_errback 

341

344 # we cache the channel for subsequent calls, this has to be

346 channel = conn.default_channel

354class QoS: 

355 """Thread safe increment/decrement of a channels prefetch_count. 

356 

357 Arguments: 

358 callback (Callable): Function used to set new prefetch count, 

359 e.g. ``consumer.qos`` or ``channel.basic_qos``. Will be called 

360 with a single ``prefetch_count`` keyword argument. 

361 initial_value (int): Initial prefetch count value.. 

362 

363 Example: 

364 >>> from kombu import Consumer, Connection 

365 >>> connection = Connection('amqp://') 

366 >>> consumer = Consumer(connection) 

367 >>> qos = QoS(consumer.qos, initial_prefetch_count=2) 

368 >>> qos.update() # set initial 

369 

370 >>> qos.value 

371 2 

372 

373 >>> def in_some_thread(): 

374 ... qos.increment_eventually() 

375 

376 >>> def in_some_other_thread(): 

377 ... qos.decrement_eventually() 

378 

379 >>> while 1: 

380 ... if qos.prev != qos.value: 

381 ... qos.update() # prefetch changed so update. 

382 

383 It can be used with any function supporting a ``prefetch_count`` keyword 

384 argument:: 

385 

386 >>> channel = connection.channel() 

387 >>> QoS(channel.basic_qos, 10) 

388 

389 

390 >>> def set_qos(prefetch_count): 

391 ... print('prefetch count now: %r' % (prefetch_count,)) 

392 >>> QoS(set_qos, 10) 

393 """ 

394 

395 prev = None 

396 

397 def __init__(self, callback, initial_value): 

398 self.callback = callback 

399 self._mutex = threading.RLock() 

400 self.value = initial_value or 0 

401 

402 def increment_eventually(self, n=1): 

403 """Increment the value, but do not update the channels QoS. 

404 

405 Note: 

406 The MainThread will be responsible for calling :meth:`update` 

407 when necessary. 

408 """ 

409 with self._mutex: 

410 if self.value: 

411 self.value = self.value + max(n, 0) 

412 return self.value 

413 

414 def decrement_eventually(self, n=1): 

415 """Decrement the value, but do not update the channels QoS. 

416 

417 Note: 

418 The MainThread will be responsible for calling :meth:`update` 

419 when necessary. 

420 """ 

421 with self._mutex: 

422 if self.value: 

423 self.value -= n 

424 if self.value < 1: 

425 self.value = 1 

426 return self.value 

427 

428 def set(self, pcount): 

429 """Set channel prefetch_count setting.""" 

430 if pcount != self.prev: 

431 new_value = pcount 

432 if pcount > PREFETCH_COUNT_MAX: 

433 logger.warning('QoS: Disabled: prefetch_count exceeds %r', 

434 PREFETCH_COUNT_MAX) 

435 new_value = 0 

436 logger.debug('basic.qos: prefetch_count->%s', new_value) 

437 self.callback(prefetch_count=new_value) 

438 self.prev = pcount 

439 return pcount 

440 

441 def update(self): 

442 """Update prefetch count with current value.""" 

443 with self._mutex: 

444 return self.set(self.value)
