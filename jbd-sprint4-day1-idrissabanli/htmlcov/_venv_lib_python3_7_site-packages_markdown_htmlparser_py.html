27# Import a copy of the html.parser lib as `htmlparser` so we can monkeypatch it. 

28# Users can still do `from html import parser` and get the default behavior. 

29spec = importlib.util.find_spec('html.parser') 

30htmlparser = importlib.util.module_from_spec(spec) 

31spec.loader.exec_module(htmlparser) 

32sys.modules['htmlparser'] = htmlparser

34# Monkeypatch HTMLParser to only accept `?>` to close Processing Instructions. 

35htmlparser.piclose = re.compile(r'\?>') 

36# Monkeypatch HTMLParser to only recognize entity references with a closing semicolon. 

37htmlparser.entityref = re.compile(r'&([a-zA-Z][-.a-zA-Z0-9]*);') 

38# Monkeypatch HTMLParser to no longer support partial entities. We are always feeding a complete block, 

39# so the 'incomplete' functionality is unnecessary. As the entityref regex is run right before incomplete, 

40# and the two regex are the same, then incomplete will simply never match and we avoid the logic within. 

41htmlparser.incomplete = htmlparser.entityref

43# Match a blank line at the start of a block of text (two newlines). 

44# The newlines may be preceded by additional whitespace. 

45blank_line_re = re.compile(r'^([ ]*\n){2}')

48class HTMLExtractor(htmlparser.HTMLParser): 

49 """ 

50 Extract raw HTML from text. 

51 

52 The raw HTML is stored in the `htmlStash` of the Markdown instance passed 

53 to `md` and the remaining text is stored in `cleandoc` as a list of strings. 

54 """

56 def __init__(self, md, *args, **kwargs): 

57 if 'convert_charrefs' not in kwargs: 

58 kwargs['convert_charrefs'] = False 

59 

60 # Block tags that should contain no content (self closing) 

61 self.empty_tags = set(['hr']) 

62 

63 # This calls self.reset 

64 super().__init__(*args, **kwargs) 

65 self.md = md

67 def reset(self): 

68 """Reset this instance. Loses all unprocessed data.""" 

69 self.inraw = False 

70 self.intail = False 

71 self.stack = [] # When inraw==True, stack contains a list of tags 

72 self._cache = [] 

73 self.cleandoc = [] 

74 super().reset()

76 def close(self): 

77 """Handle any buffered data.""" 

78 super().close() 

79 if len(self.rawdata): 

80 # Temp fix for https://bugs.python.org/issue41989 

81 # TODO: remove this when the bug is fixed in all supported Python versions. 

82 if self.convert_charrefs and not self.cdata_elem: # pragma: no cover 

83 self.handle_data(htmlparser.unescape(self.rawdata)) 

84 else: 

85 self.handle_data(self.rawdata) 

86 # Handle any unclosed tags. 

87 if len(self._cache): 

88 self.cleandoc.append(self.md.htmlStash.store(''.join(self._cache))) 

89 self._cache = []

91 @property 

92 def line_offset(self): 

93 """Returns char index in self.rawdata for the start of the current line. """ 

94 if self.lineno > 1: 

95 return re.match(r'([^\n]*\n){{{}}}'.format(self.lineno-1), self.rawdata).end() 

96 return 0

98 def at_line_start(self): 

99 """ 

100 Returns True if current position is at start of line. 

101 

102 Allows for up to three blank spaces at start of line. 

103 """ 

104 if self.offset == 0: 

105 return True 

106 if self.offset > 3: 

107 return False 

108 # Confirm up to first 3 chars are whitespace 

109 return self.rawdata[self.line_offset:self.line_offset + self.offset].strip() == ''

111 def get_endtag_text(self, tag): 

112 """ 

113 Returns the text of the end tag. 

114 

115 If it fails to extract the actual text from the raw data, it builds a closing tag with `tag`. 

116 """ 

117 # Attempt to extract actual tag from raw source text 

118 start = self.line_offset + self.offset 

119 m = htmlparser.endendtag.search(self.rawdata, start) 

120 if m: 

121 return self.rawdata[start:m.end()] 

122 else: # pragma: no cover 

123 # Failed to extract from raw data. Assume well formed and lowercase. 

124 return '</{}>'.format(tag)

126 def handle_starttag(self, tag, attrs): 

127 # Handle tags that should always be empty and do not specify a closing tag 

128 if tag in self.empty_tags: 

129 self.handle_startendtag(tag, attrs) 

130 return 

131 

132 if self.md.is_block_level(tag) and (self.intail or (self.at_line_start() and not self.inraw)): 

133 # Started a new raw block. Prepare stack. 

134 self.inraw = True 

135 self.cleandoc.append('\n') 

136 

137 text = self.get_starttag_text() 

138 if self.inraw: 

139 self.stack.append(tag) 

140 self._cache.append(text) 

141 else: 

142 self.cleandoc.append(text) 

143 if tag in self.CDATA_CONTENT_ELEMENTS: 

144 # This is presumably a standalone tag in a code span (see #1036). 

145 self.clear_cdata_mode()

147 def handle_endtag(self, tag): 

148 text = self.get_endtag_text(tag) 

149 

150 if self.inraw: 

151 self._cache.append(text) 

152 if tag in self.stack: 

153 # Remove tag from stack 

154 while self.stack: 

155 if self.stack.pop() == tag: 

156 break 

157 if len(self.stack) == 0: 

158 # End of raw block. 

159 if blank_line_re.match(self.rawdata[self.line_offset + self.offset + len(text):]): 

160 # Preserve blank line and end of raw block. 

161 self._cache.append('\n') 

162 else: 

163 # More content exists after endtag. 

164 self.intail = True 

165 # Reset stack. 

166 self.inraw = False 

167 self.cleandoc.append(self.md.htmlStash.store(''.join(self._cache))) 

168 # Insert blank line between this and next line. 

169 self.cleandoc.append('\n\n') 

170 self._cache = [] 

171 else: 

172 self.cleandoc.append(text)

174 def handle_data(self, data): 

175 if self.intail and '\n' in data: 

176 self.intail = False 

177 if self.inraw: 

178 self._cache.append(data) 

179 else: 

180 self.cleandoc.append(data)

182 def handle_empty_tag(self, data, is_block): 

183 """ Handle empty tags (`<data>`). """ 

184 if self.inraw or self.intail: 

185 # Append this to the existing raw block 

186 self._cache.append(data) 

187 elif self.at_line_start() and is_block: 

188 # Handle this as a standalone raw block 

189 if blank_line_re.match(self.rawdata[self.line_offset + self.offset + len(data):]): 

190 # Preserve blank line after tag in raw block. 

191 data += '\n' 

192 else: 

193 # More content exists after tag. 

194 self.intail = True 

195 item = self.cleandoc[-1] if self.cleandoc else '' 

196 # If we only have one newline before block element, add another 

197 if not item.endswith('\n\n') and item.endswith('\n'): 

198 self.cleandoc.append('\n') 

199 self.cleandoc.append(self.md.htmlStash.store(data)) 

200 # Insert blank line between this and next line. 

201 self.cleandoc.append('\n\n') 

202 else: 

203 self.cleandoc.append(data)

205 def handle_startendtag(self, tag, attrs): 

206 self.handle_empty_tag(self.get_starttag_text(), is_block=self.md.is_block_level(tag))

223 def unknown_decl(self, data): 

224 end = ']]>' if data.startswith('CDATA[') else ']>'

237 def parse_starttag(self, i): # pragma: no cover 

238 self.__starttag_text = None 

239 endpos = self.check_for_whole_start_tag(i) 

240 if endpos < 0: 

241 return endpos 

242 rawdata = self.rawdata 

243 self.__starttag_text = rawdata[i:endpos] 

244 

245 # Now parse the data between i+1 and j into a tag and attrs 

246 attrs = [] 

247 match = htmlparser.tagfind_tolerant.match(rawdata, i+1) 

248 assert match, 'unexpected call to parse_starttag()' 

249 k = match.end() 

250 self.lasttag = tag = match.group(1).lower() 

251 while k < endpos: 

252 m = htmlparser.attrfind_tolerant.match(rawdata, k) 

253 if not m: 

254 break 

255 attrname, rest, attrvalue = m.group(1, 2, 3) 

256 if not rest: 

257 attrvalue = None 

258 elif attrvalue[:1] == '\'' == attrvalue[-1:] or \ 

259 attrvalue[:1] == '"' == attrvalue[-1:]: # noqa: E127 

260 attrvalue = attrvalue[1:-1] 

261 if attrvalue: 

262 attrvalue = htmlparser.unescape(attrvalue) 

263 attrs.append((attrname.lower(), attrvalue)) 

264 k = m.end() 

265 

266 end = rawdata[k:endpos].strip() 

267 if end not in (">", "/>"): 

268 lineno, offset = self.getpos() 

269 if "\n" in self.__starttag_text: 

270 lineno = lineno + self.__starttag_text.count("\n") 

271 offset = len(self.__starttag_text) \ 

272 - self.__starttag_text.rfind("\n") # noqa: E127 

273 else: 

274 offset = offset + len(self.__starttag_text) 

275 self.handle_data(rawdata[i:endpos]) 

276 return endpos 

277 if end.endswith('/>'): 

278 # XHTML-style empty tag: <span attr="value" /> 

279 self.handle_startendtag(tag, attrs) 

280 else: 

281 # *** set cdata_mode first so we can override it in handle_starttag (see #1036) *** 

282 if tag in self.CDATA_CONTENT_ELEMENTS: 

283 self.set_cdata_mode(tag) 

284 self.handle_starttag(tag, attrs) 

285 return endpos
